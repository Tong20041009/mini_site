# -*- coding: utf-8 -*- """ 跳动的爱心（Tkinter 版 · 稳定优化） - 修复：log(0) 与除零风险（加入 _EPS） - 用整数画布中心，坐标更稳 - 其余逻辑与你的版本一致，保留中心祝福文字 """ import random from math import sin, cos, pi, log from tkinter import Tk, Canvas # ===== 可自定义区域 ===== CANVAS_WIDTH = 640 CANVAS_HEIGHT = 480 IMAGE_ENLARGE = 11 # 心形整体缩放 HEART_COLOR = "#e8e184" # 爱心颜色 WINDOWS_TITLE = "爱心" # 窗口标题 FRAME_INTERVAL_MS = 160 # 每帧间隔（越小越快） # ====================== # 画布中心用整数更稳 CANVAS_CENTER_X = CANVAS_WIDTH // 2 CANVAS_CENTER_Y = CANVAS_HEIGHT // 2 # 数值稳定常数 _EPS = 1e-8 def heart_function(t, shrink_ratio: float = IMAGE_ENLARGE): """心形参数方程（y 取反使心朝上），返回整数像素坐标""" x = 16 * (sin(t) ** 3) y = -(13 * cos(t) - 5 * cos(2 * t) - 2 * cos(3 * t) - cos(4 * t)) x *= shrink_ratio; y *= shrink_ratio x += CANVAS_CENTER_X; y += CANVAS_CENTER_Y return int(x), int(y) def scatter_inside(x, y, beta=0.15): """向中心对数分布收缩，避免 log(0)""" ratio_x = -beta * log(random.random() + _EPS) ratio_y = -beta * log(random.random() + _EPS) dx = ratio_x * (x - CANVAS_CENTER_X) dy = ratio_y * (y - CANVAS_CENTER_Y) return x - dx, y - dy def shrink(x, y, ratio): """按距离收缩（带负号表示朝中心），分母加 _EPS 防爆""" r2 = (x - CANVAS_CENTER_X) ** 2 + (y - CANVAS_CENTER_Y) ** 2 + _EPS force = -1 / (r2 ** 1.6) dx = ratio * force * (x - CANVAS_CENTER_X) dy = ratio * force * (y - CANVAS_CENTER_Y) return x - dx, y - dy def curve(p): """心跳节奏函数（-~+之间的平滑周期值）""" return 2 * (2 * sin(4 * p)) / (2 * pi) class Heart: def __init__(self, generate_frame=20): self._points = set() self._edge_diffusion_points = set() self._center_diffusion_points = set() self.all_points = {} self.build(2000) self.random_halo = 1000 self.generate_frame = generate_frame for frame in range(generate_frame): self.calc(frame) def build(self, number): # 边界采样 for _ in range(number): t = random.uniform(0, 2 * pi) self._points.add(heart_function(t)) # 边缘扩散 for _x, _y in list(self._points): for _ in range(3): self._edge_diffusion_points.add(scatter_inside(_x, _y, 0.05)) # 中心扩散 point_list = list(self._points) for _ in range(4000): x, y = random.choice(point_list) self._center_diffusion_points.add(scatter_inside(x, y, 0.17)) @staticmethod def calc_position(x, y, ratio): r2 = (x - CANVAS_CENTER_X) ** 2 + (y - CANVAS_CENTER_Y) ** 2 + _EPS force = 1 / (r2 ** 0.9) dx = ratio * force * (x - CANVAS_CENTER_X) + random.randint(-1, 1) dy = ratio * force * (y - CANVAS_CENTER_Y) + random.randint(-1, 1) return x - dx, y - dy def calc(self, generate_frame): ratio = 10 * curve(generate_frame / 10 * pi) halo_radius = int(4 + 6 * (1 + curve(generate_frame / 10 * pi))) halo_number = int(3000 + 4000 * abs(curve(generate_frame / 10 * pi) ** 2)) all_points = [] heart_halo_point = set() # 光环 for _ in range(halo_number): t = random.uniform(0, 2 * pi) x, y = heart_function(t, shrink_ratio=11.6) x, y = shrink(x, y, halo_radius) if (x, y) not in heart_halo_point: heart_halo_point.add((x, y)) x += random.randint(-14, 14) y += random.randint(-14, 14) size = random.choice((1, 2, 2)) all_points.append((x, y, size)) # 轮廓 + 扩散 for x, y in self._points: x, y = self.calc_position(x, y, ratio) all_points.append((x, y, random.randint(1, 3))) for x, y in self._edge_diffusion_points: x, y = self.calc_position(x, y, ratio) all_points.append((x, y, random.randint(1, 2))) for x, y in self._center_diffusion_points: x, y = self.calc_position(x, y, ratio) all_points.append((x, y, random.randint(1, 2))) self.all_points[generate_frame] = all_points def render(self, render_canvas: Canvas, render_frame: int): pts = self.all_points[render_frame % self.generate_frame] for x, y, size in pts: render_canvas.create_rectangle(x, y, x + size, y + size, width=0, fill=HEART_COLOR) def draw(main: Tk, render_canvas: Canvas, render_heart: Heart, render_frame=0): render_canvas.delete('all') render_heart.render(render_canvas, render_frame) # 中心祝福文字 render_canvas.create_text( CANVAS_CENTER_X, CANVAS_CENTER_Y, text=HEART_CENTER_TEXT, fill=HEART_CENTER_TEXT_COLOR, font=('Arial', 20, 'bold') ) main.after(FRAME_INTERVAL_MS, draw, main, render_canvas, render_heart, render_frame + 1) if __name__ == '__main__': root = Tk() root.title(WINDOWS_TITLE) canvas = Canvas(root, bg='black', height=CANVAS_HEIGHT, width=CANVAS_WIDTH, highlightthickness=0) canvas.pack() heart = Heart() draw(root, canvas, heart) root.mainloop()
