# -*- coding: utf-8 -*- """ 跳动的爱心（Tkinter 版） - 心形参数方程 + 扩散采样 + 心跳扰动 - 已修正：心形正方向、cos(4*t) 误写、括号/幅度等细节 """ import random from math import sin, cos, pi, log from tkinter import Tk, Canvas # 画布与配色 CANVAS_WIDTH = 640 CANVAS_HEIGHT = 480 CANVAS_CENTER_X = CANVAS_WIDTH // 2 CANVAS_CENTER_Y = CANVAS_HEIGHT // 2 IMAGE_ENLARGE = 11 HEART_COLOR = "#0000FF" # 按你截图：蓝色。想改粉色可用 "#ff2b7f" # ---------- 几何/采样工具 ---------- def heart_function(t, shrink_ratio: float = IMAGE_ENLARGE): """心形参数方程，返回画布坐标（整数像素） 取反 y 以保证心形朝上（屏幕坐标 y 轴向下） """ x = 15 * (sin(t) ** 3) y = -(13 * cos(t) - 4 * cos(2 * t) - 2 * cos(3 * t) - cos(4 * t)) # ← 注意是 cos(4*t) x *= shrink_ratio y *= shrink_ratio x += CANVAS_CENTER_X y += CANVAS_CENTER_Y return int(x), int(y) def scatter_inside(x, y, beta=0.15): """朝中心随机“收缩”一点（对数分布，聚拢更自然）""" ratio_x = - beta * log(random.random() + 1e-8) ratio_y = - beta * log(random.random() + 1e-8) dx = ratio_x * (x - CANVAS_CENTER_X) dy = ratio_y * (y - CANVAS_CENTER_Y) return x - dx, y - dy def shrink(x, y, ratio): """以中心为基准收缩（带轻微噪声），ratio 越大收缩越强""" # 与截图一致：距离越远力越小；负号让位移朝向中心 force = -1 / (((x - CANVAS_CENTER_X) ** 2 + (y - CANVAS_CENTER_Y) ** 2) ** 0.6 + 1e-6) dx = ratio * force * (x - CANVAS_CENTER_X) dy = ratio * force * (y - CANVAS_CENTER_Y) return x - dx, y - dy def curve(p): """心跳节奏函数（平滑周期 -1..1）""" return sin(2 * pi * p) # ---------- 粒子心形 ---------- class Heart: def __init__(self, generate_frame: int = 20): self._points = set() self._edge_diffusion_points = set() self._center_diffusion_points = set() self.all_points = {} # 每帧渲染点 [(x,y,size), ...] self.build(2000) # 采样基础点 self.random_halo = 1000 self.generate_frame = generate_frame for frame in range(generate_frame): self.calc(frame) def build(self, number: int): """采样心形边界 + 扩散""" # 边界采样 for _ in range(number): t = random.uniform(0, 2 * pi) x, y = heart_function(t) self._points.add((x, y)) # 边缘扩散 for _x, _y in list(self._points): for _ in range(3): x, y = scatter_inside(_x, _y, 0.05) self._edge_diffusion_points.add((int(x), int(y))) # 中心扩散 point_list = list(self._points) for _ in range(4000): x, y = random.choice(point_list) x, y = scatter_inside(x, y, 0.17) self._center_diffusion_points.add((int(x), int(y))) @staticmethod def calc_position(x, y, ratio): """按心跳强度把点向外/向内轻推一点""" force = 1 / (((x - CANVAS_CENTER_X) ** 2 + (y - CANVAS_CENTER_Y) ** 2) ** 0.52 + 1e-6) x -= ratio * force * (x - CANVAS_CENTER_X) + random.randint(-1, 1) y -= ratio * force * (y - CANVAS_CENTER_Y) + random.randint(-1, 1) return x, y def calc(self, generate_frame: int): """计算一帧需要渲染的点""" ratio = 12 + 12 * curve(generate_frame / 10) # 心跳强度（可调） halo_radius = int(4 + 6 * (1 + curve(generate_frame / 10))) halo_number = int(3000 + 4000 * abs(curve(generate_frame / 10)) ** 2) all_points = [] # 光晕 heart_halo_point = set() for _ in range(halo_number): t = random.uniform(0, 2 * pi) x, y = heart_function(t, shrink_ratio=11.6) x, y = shrink(x, y, halo_radius) if (x, y) not in heart_halo_point: heart_halo_point.add((x, y)) x += random.randint(-14, 14) y += random.randint(-14, 14) size = random.choice((1, 2, 2)) all_points.append((x, y, size)) # 原始边界 for x, y in self._points: x, y = self.calc_position(x, y, ratio) size = random.randint(1, 2) all_points.append((x, y, size)) # 边缘扩散 for x, y in self._edge_diffusion_points: x, y = self.calc_position(x, y, ratio) size = random.randint(1, 2) all_points.append((x, y, size)) # 中心扩散 for x, y in self._center_diffusion_points: x, y = self.calc_position(x, y, ratio) size = random.randint(1, 2) all_points.append((x, y, size)) self.all_points[generate_frame] = all_points def render(self, render_canvas: Canvas, render_frame: int): """画出一帧""" for x, y, size in self.all_points[render_frame % self.generate_frame]: render_canvas.create_rectangle(x, y, x + size, y + size, width=0, fill=HEART_COLOR) # ---------- 动画循环 ---------- def draw(main: Tk, render_canvas: Canvas, render_heart: Heart, render_frame=0): render_canvas.delete('all') render_heart.render(render_canvas, render_frame) main.after(160, draw, main, render_canvas, render_heart, render_frame + 1) # 每帧 160ms（≈6FPS） if __name__ == '__main__': root = Tk() root.title("Beating Heart - Tkinter") canvas = Canvas(root, bg='black', height=CANVAS_HEIGHT, width=CANVAS_WIDTH, highlightthickness=0) canvas.pack() heart = Heart() draw(root, canvas, heart) root.mainloop()
