# -*- coding: utf-8 -*- """ 用 Tkinter 画“跳动的爱心粒子” 原理：心形参数方程 + 扩散采样 + 每帧微小位移 + 帧循环 """ import random from math import sin, cos, pi, log from tkinter import Tk, Canvas # 画布与配色 CANVAS_WIDTH = 640 CANVAS_HEIGHT = 480 CANVAS_CENTER_X = CANVAS_WIDTH // 2 CANVAS_CENTER_Y = CANVAS_HEIGHT // 2 IMAGE_ENLARGE = 11 # 心形整体放大倍数 HEART_COLOR = "#ff2121" # 你截图里的主色 # ------------ 几何/采样工具 ------------ def heart_function(t, shrink_ratio: float = IMAGE_ENLARGE): """心形参数方程，返回画布坐标（整数像素）""" # 标准心形（单位坐标系） x = 16 * (sin(t) ** 3) y = 13 * cos(t) - 5 * cos(2 * t) - 2 * cos(3 * t) - cos(4 * t) # 放大 + 画布平移 x *= shrink_ratio y *= shrink_ratio x += CANVAS_CENTER_X y += CANVAS_CENTER_Y return int(x), int(y) def scatter_inside(x, y, beta=0.15): """把点朝中心“随机拉近”一点（对数分布更聚拢）""" ratio_x = -beta * log(random.random() + 1e-8) ratio_y = -beta * log(random.random() + 1e-8) dx = ratio_x * (x - CANVAS_CENTER_X) dy = ratio_y * (y - CANVAS_CENTER_Y) return x - dx, y - dy def shrink(x, y, ratio): """把点以中心为基准收缩一些（带一点噪声）""" force = 1 / (((x - CANVAS_CENTER_X) ** 2 + (y - CANVAS_CENTER_Y) ** 2) ** 0.6 + 1e-6) dx = ratio * force * (x - CANVAS_CENTER_X) + random.randint(-1, 1) dy = ratio * force * (y - CANVAS_CENTER_Y) + random.randint(-1, 1) return x - dx, y - dy def curve(p): """一个平滑的周期函数，用来做心跳（-1~1）""" # 你图里是 2*(2*sin(4πp))/(2π) 的变体，我们做个等价的平滑波 return sin(2 * pi * p) # ------------ 粒子心形类 ------------ class Heart: def __init__(self, generate_frame: int = 20): # 原始心形点集 & 扩散点集 self._points = set() self._edge_diffusion_points = set() self._center_diffusion_points = set() self.all_points = {} # 每帧需要渲染的点 [(x,y,size), ...] self.random_halo = 1000 # 光晕随机种子 self.generate_frame = generate_frame self.build(2000) # 先采样一批基础点 for f in range(generate_frame): self.calc(f) def build(self, number: int): """采样出基础心形 + 向心/边缘扩散""" # 1) 采样心形边界 for _ in range(number): t = random.uniform(0, 2 * pi) x, y = heart_function(t) self._points.add((x, y)) # 2) 边缘扩散：每个边界点再抖三下 for _x, _y in list(self._points): for _ in range(3): x, y = scatter_inside(_x, _y, 0.05) self._edge_diffusion_points.add((int(x), int(y))) # 3) 中心扩散：再从所有点里多撒一些到中心 point_list = list(self._points) for _ in range(4000): x, y = random.choice(point_list) x, y = scatter_inside(x, y, 0.17) self._center_diffusion_points.add((int(x), int(y))) def calc_position(self, x, y, ratio): """把点向外/向内轻推一点（随帧而变）""" force = 1 / (((x - CANVAS_CENTER_X) ** 2 + (y - CANVAS_CENTER_Y) ** 2) ** 0.52 + 1e-6) dx = ratio * force * (x - CANVAS_CENTER_X) + random.randint(-1, 1) dy = ratio * force * (y - CANVAS_CENTER_Y) + random.randint(-1, 1) return x - dx, y - dy def calc(self, generate_frame: int): """按帧生成一份点列表，存入 all_points""" ratio = 10 + curve(generate_frame / 10) * 10 # 心跳缩放强度（随帧变化） # 光晕：在心形边缘附近围一圈发光点 halo_radius = int(4 + 6 * (1 + abs(curve(generate_frame / 10)))) halo_number = int(3000 + 4000 * (abs(curve(generate_frame / 10)) ** 2)) all_points = [] heart_halo_point = set() for _ in range(halo_number): t = random.uniform(0, 2 * pi) x, y = heart_function(t, shrink_ratio=11.6) x, y = shrink(x, y, halo_radius) if (x, y) not in heart_halo_point: heart_halo_point.add((x, y)) x += random.randint(-14, 14) y += random.randint(-14, 14) size = random.choice((1, 2, 2)) all_points.append((x, y, size)) # 原始边界点 for x, y in self._points: x, y = self.calc_position(x, y, ratio) size = random.randint(1, 2) all_points.append((x, y, size)) # 边缘扩散 for x, y in self._edge_diffusion_points: x, y = self.calc_position(x, y, ratio) size = random.randint(1, 2) all_points.append((x, y, size)) # 中心扩散 for x, y in self._center_diffusion_points: x, y = self.calc_position(x, y, ratio) size = random.randint(1, 2) all_points.append((x, y, size)) self.all_points[generate_frame] = all_points def render(self, render_canvas: Canvas, render_frame: int): """把当前帧的所有点画到画布上""" points = self.all_points[render_frame % self.generate_frame] for x, y, size in points: render_canvas.create_rectangle(x, y, x + size, y + size, width=0, fill=HEART_COLOR) # ------------ 动画循环 ------------ def draw(main: Tk, render_canvas: Canvas, render_heart: Heart, render_frame=0): render_canvas.delete('all') render_heart.render(render_canvas, render_frame) # 160ms 左右一帧（约 6FPS），你可以改成 60~120 更流畅/更耗电 main.after(160, draw, main, render_canvas, render_heart, render_frame + 1) if __name__ == '__main__': root = Tk() root.title("Beating Heart - Tkinter") canvas = Canvas(root, bg='black', height=CANVAS_HEIGHT, width=CANVAS_WIDTH, highlightthickness=0) canvas.pack() heart = Heart(generate_frame=20) draw(root, canvas, heart) root.mainloop()
